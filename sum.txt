-- =====================================================================
-- PostgreSQL equivalent of Oracle VENDEE.SUM_UP package
-- Converted from Oracle PL/SQL to Azure PostgreSQL (PL/pgSQL)
--
-- Conversion notes:
--   - PostgreSQL has no PACKAGE concept; procedures live directly in schema
--   - SYSTIMESTAMP                    → CURRENT_TIMESTAMP
--   - sequence.NEXTVAL FROM DUAL      → NEXTVAL('schema.sequence_name')
--   - SELECT ... FROM DUAL            → SELECT ... (no DUAL in PostgreSQL)
--   - BULK COLLECT INTO               → ARRAY_AGG(...) INTO array_variable
--   - TYPE ... IS TABLE OF t%TYPE     → native array type (BIGINT[], TEXT[])
--   - IN (SELECT * FROM TABLE(arr))   → = ANY(arr)
--   - FOR I IN 1..arr.COUNT / arr(I)  → FOREACH item IN ARRAY arr
--   - VARCHAR2                        → VARCHAR / TEXT
--   - NUMBER                          → NUMERIC
--   - VARCHAR2_ARRAY (Oracle custom)  → TEXT[]
-- =====================================================================

CREATE OR REPLACE PROCEDURE vendee.sum_up_office(
    IN  i_customer_code              VARCHAR,
    IN  i_office_code                VARCHAR,
    IN  i_sum_up_date                DATE,
    IN  i_sum_up_execute_date        TIMESTAMP,
    IN  i_sum_up_execute_person      VARCHAR,
    IN  i_sum_up_before_check_status NUMERIC,
    INOUT o_sum_up_management_id     NUMERIC DEFAULT NULL
)
LANGUAGE plpgsql
AS $$
DECLARE
    v_system_timestamp  TIMESTAMP;
    sumgmt_id           NUMERIC;
    salesdata           BIGINT[];   -- replaces: TYPE SALESLIST IS TABLE OF SALES.ID%TYPE
    sales_detail_id     NUMERIC;
    cnums2              TEXT[];     -- replaces: VARCHAR2_ARRAY
    v_sales_id          BIGINT;
BEGIN

    -- システム時間と締処理管理IDの採番
    v_system_timestamp := CURRENT_TIMESTAMP;

    SELECT NEXTVAL('vendee.sum_up_management_seq') INTO sumgmt_id;

    -- OUTパラメータに値を設定
    o_sum_up_management_id := sumgmt_id;

    -- 締処理情報管理を登録する
    INSERT INTO sum_up_management
           ( id
           , customer_code
           , customer_office_code
           , sum_up_date
           , sum_up_execute_date
           , sum_up_execute_person
           , cancel_date          -- NULL
           , cancel_person        -- NULL
           , is_cancel            -- 0
           , entrust_customer_office_code  -- NULL
           , available            -- 1
           , create_date
           , create_user
           , update_date
           , update_user
           , remark )
    VALUES ( sumgmt_id
           , i_customer_code
           , i_office_code
           , i_sum_up_date
           , i_sum_up_execute_date
           , i_sum_up_execute_person
           , NULL
           , NULL
           , 0
           , NULL
           , 1
           , v_system_timestamp
           , 'BGJOB'
           , v_system_timestamp
           , 'BGJOB'
           , '1'  -- remark: 締処理中「1」
           );

    -- コミュニケーション番号取得
    -- Oracle: BULK COLLECT INTO → PostgreSQL: ARRAY_AGG INTO
    SELECT ARRAY_AGG(r.rw_communication_number)
      INTO cnums2
      FROM m_rw r
     INNER JOIN m_vending_machine v ON r.vending_machine_id = v.id
     INNER JOIN m_customer_office c ON v.customer_office_id = c.id
     WHERE c.customer_code        = i_customer_code
       AND c.customer_office_code = i_office_code
       AND r.available = 1
       AND v.available = 1
       AND c.available = 1;

    -- 販売実績データ取得
    -- Oracle: IN (SELECT * FROM TABLE(cnums2)) → PostgreSQL: = ANY(cnums2)
    -- Oracle: DATE + 1 (adds 1 day) works the same in PostgreSQL
    SELECT ARRAY_AGG(s.id)
      INTO salesdata
      FROM sales s
     WHERE s.rw_communication_number = ANY(cnums2)
       AND s.sum_up_select_sales_date < i_sum_up_date + 1
       AND s.is_sum_up_executed = 0
       AND s.available = 1;

    -- 締処理明細テーブルへの登録
    -- Oracle: FOR I IN 1..SALESDATA.COUNT / SALESDATA(I)
    -- PostgreSQL: FOREACH item IN ARRAY salesdata
    IF salesdata IS NOT NULL THEN
        FOREACH v_sales_id IN ARRAY salesdata LOOP

            SELECT NEXTVAL('vendee.sum_up_detail_seq') INTO sales_detail_id;

            INSERT INTO sum_up_detail
            VALUES ( sales_detail_id
                   , sumgmt_id
                   , v_sales_id
                   , 1                   -- available
                   , v_system_timestamp  -- create_date
                   , 'BGJOB'             -- create_user
                   , v_system_timestamp  -- update_date
                   , 'BGJOB'             -- update_user
                   , NULL                -- remark
                   );

        END LOOP;
    END IF;

    -- 販売管理テーブルの更新
    UPDATE sales s
       SET is_sum_up_executed = 1
         , update_user        = 'BGJOB'
         , update_date        = v_system_timestamp
     WHERE EXISTS (
        SELECT 1
          FROM sum_up_detail sud
         WHERE sud.sales_id             = s.id
           AND sud.sum_up_management_id = sumgmt_id
     );

    -- 締処理状況テーブルの更新 (2011.01.11 kirie add)
    UPDATE sum_up_status
       SET sum_up_status              = 0
         , sum_up_before_check_status = i_sum_up_before_check_status  -- 2011.07.22 otobe add
         , update_user                = 'BGJOB'
         , update_date                = v_system_timestamp
     WHERE customer_code        = i_customer_code
       AND customer_office_code = i_office_code
       AND sum_up_status        = 1;

END;
$$;
